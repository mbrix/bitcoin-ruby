def get_tx(hash)
  url = "http://blockexplorer.com/%srawtx/%s" % [$testnet ? 'testnet/' : '',  hash]
  json = open(url).read
  Bitcoin::Protocol::Tx.from_json(json)
rescue Exception
  nil
end
