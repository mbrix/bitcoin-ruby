    Load keys from file. If file is emty this will generate a new key and store it, creating the file.

    # File lib/bitcoin/wallet/keystore.rb, line 57
def load_keys
  if File.exist?(@config[:file])
    data = JSON.load(File.read(@config[:file]))
    data.map {|a, k| @keys[a] = Bitcoin::Key.from_base58(k)}
  else
    new_key; save_keys
  end
end

def save_keys
  File.open(@config[:file], 'w') do |file|
    file.write(Hash[@keys.map {|a, k| [a, k.to_base58]}].to_json)
  end
end
