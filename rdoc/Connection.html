def on_request(req)
  result = case req['method']
  when 'store_block'
             p :req
             store_block(*req['params'].map {|b| Bitcoin::P::Block.new(b.htb) })
  end

  respond({id: req['id'], result: result})
  
rescue
  p $!
end

def receive_data data
  EM.defer do
    @buf.extract(data).each do |packet|
      on_request(JSON.load(packet))
    end
  end
end

def store_block *blks
  p *blks.size
  @server.lock.synchronize do
    blks.map do |blk|
      res = @server.store.store_block(blk)
      p res
      res
    end
  end
end
